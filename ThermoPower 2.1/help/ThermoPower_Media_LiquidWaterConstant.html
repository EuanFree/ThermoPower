Modelica definition

redeclare model extends BaseProperties "Base properties of medium"
equation 
  // h = cp0*(T-T0)+h0;
  T = T0 + (h-h0)/cp0;
  d = rho0*(1-beta0*(T-T0));
  u = h;
  R=1;
  MM=0.018;
  state.T=T;
end BaseProperties;
