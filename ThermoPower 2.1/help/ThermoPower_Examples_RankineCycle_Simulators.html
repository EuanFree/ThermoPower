ThermoPower.Examples.RankineCycle.Simulators.OpenLoop

Information



This model allows to simulate the open loop transients.



Modelica definition


model OpenLoop
  package FlueGas = ThermoPower.Media.FlueGas "Flue gas model";
  package Water = ThermoPower.Water.StandardWater "Fluid model";
  Modelica.Blocks.Sources.Ramp gasFlowRate(offset=500, height=0);
  Modelica.Blocks.Sources.Ramp gasTemperature(offset=750,
    height=10,
    duration=0,
    startTime=2000);
  ThermoPower.Examples.RankineCycle.Plant plant(
                   SSInit=true,
    redeclare package FlueGas = FlueGas,
    redeclare package Water = Water,
    evaporator(water(initOpt=ThermoPower.Choices.Init.Options.noInit)));
  Modelica.Blocks.Sources.Ramp nPump(
    offset=1500,
    startTime=2000,
    height=0,
    duration=2);
equation 
  connect(gasTemperature.y, plant.gasTemperature);
  connect(plant.gasFlowRate, gasFlowRate.y);
  connect(plant.nPump, nPump.y);
end OpenLoop;







ThermoPower.Examples.RankineCycle.Simulators.ClosedLoop



Information




This model simulates a simple continuous-time control system for the steam power plant.

The generated power and the evaporator void fraction are controlled to the set point by PI controllers with anti-windup.




Modelica definition


model ClosedLoop
  package FlueGas = ThermoPower.Media.FlueGas "Flue gas model";
  package Water = ThermoPower.Water.StandardWater "Fluid model";
  Modelica.Blocks.Sources.Ramp gasFlowRate(offset=500, height=0);
  ThermoPower.Examples.RankineCycle.Plant plant(redeclare package FlueGas =
        FlueGas, redeclare package Water = Water,
    SSInit=true);
  Modelica.Blocks.Sources.Step voidFractionSetPoint(
    offset=0.2,
    height=0,
    startTime=0);
  PID voidFractionController(
    CSmin=500,
    PVmin=0.1,
    PVmax=0.9,
    CSmax=2500,
    Ti=1000,
    Kp=-0.7,
    PVstart=0.1,
    CSstart=0.5,
    steadyStateInit=true);
  Modelica.Blocks.Sources.Ramp powerSetPoint(
    offset=56.8e6,
    height=-56.8e6*0.35,
    duration=450,
    startTime=500);
  PID powerController(
    PVmin=40e6,
    PVmax=65e6,
    CSmin=500,
    CSmax=800,
    Ti=15,
    Kp=0.5,
    CSstart=0.8,
    steadyStateInit=true);
equation 
  connect(plant.gasFlowRate, gasFlowRate.y);
  connect(voidFractionController.SP, voidFractionSetPoint.y);
  connect(voidFractionController.CS, plant.nPump);
  connect(voidFractionController.PV, plant.voidFraction);
  connect(powerSetPoint.y, powerController.SP);
  connect(powerController.PV, plant.generatedPower);
  connect(plant.gasTemperature, powerController.CS);
end ClosedLoop;
